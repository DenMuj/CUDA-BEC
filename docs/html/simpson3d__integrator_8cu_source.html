73 double integrate(double hx, double hy, double hz, double *h_f,
74 long Nx, long Ny, long Nz) {
75 double total_sum = 0.0;
76
77 // Process the volume in tiles along the Z direction
78 for (long z_start = 0; z_start < Nz; z_start += tile_size_z) {
79 // Calculate the actual size of this tile (last tile might be smaller)
80 long current_tile_z = std::min(tile_size_z, Nz - z_start);
81 long tile_points = Nx * Ny * current_tile_z;
82
83 // Copy this tile's data from host to device
84 cudaMemcpy(d_f, h_f + z_start * Nx * Ny,
85 tile_points * sizeof(double), cudaMemcpyHostToDevice);
86
87 // Reset the partial sum for this tile
88 cudaMemset(d_partial_sum, 0, sizeof(double));
89
90 // Launch kernel for this tile
91 launchSimpson3DKernel(d_f, d_partial_sum, Nx, Ny, Nz,
92 tile_size_z, z_start, current_tile_z);
93
94 // Check for kernel launch errors
95 cudaError_t err = cudaGetLastError();
96 if (err != cudaSuccess) {
97 std::cerr << "Kernel launch error: " << cudaGetErrorString(err) << std::endl;
98 return 0.0;
99 }
100
101 // Check for kernel execution errors
102 err = cudaGetLastError();
103 if (err != cudaSuccess) {
104 std::cerr << "Kernel execution error: " << cudaGetErrorString(err) << std::endl;
105 return 0.0;
106 }
107
108 // Use async transfer with pinned memory for better performance
109 cudaMemcpyAsync(h_tile_sum_pinned, d_partial_sum, sizeof(double),
110 cudaMemcpyDeviceToHost, stream);
111
112 // Wait for the async transfer to complete
113 cudaStreamSynchronize(stream);
114
115 // Accumulate the result
116 total_sum += *h_tile_sum_pinned;
117 }
118
119 // Apply Simpson's rule scaling factor
120 return total_sum * hx * hy * hz / 27.0;
121 }

184 void setTileSize(long new_tile_size) {

186 throw std::invalid_argument("Tile size must be positive");
187 }
188
189 tile_size_z = new_tile_size;
190 // Reallocate if new size is larger

192 if (new_max_points > max_tile_points) {
193 cudaFree(d_f);

196
