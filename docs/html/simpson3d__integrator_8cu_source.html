65 double integrate(double hx, double hy, double hz, double *h_f,
66 long Nx, long Ny, long Nz) {
67 double total_sum = 0.0;
68
69 // Process the volume in tiles along the Z direction
70 for (long z_start = 0; z_start < Nz; z_start += tile_size_z) {
71 // Calculate the actual size of this tile (last tile might be smaller)
72 long current_tile_z = std::min(tile_size_z, Nz - z_start);
73 long tile_points = Nx * Ny * current_tile_z;
74
75 // Copy this tile's data from host to device
76 cudaMemcpy(d_f, h_f + z_start * Nx * Ny,
77 tile_points * sizeof(double), cudaMemcpyHostToDevice);
78
79 // Reset the partial sum for this tile
80 cudaMemset(d_partial_sum, 0, sizeof(double));
81
82 // Launch kernel for this tile
83 launchSimpson3DKernel(d_f, d_partial_sum, Nx, Ny, Nz,
84 tile_size_z, z_start, current_tile_z);
85
86 // Synchronize to ensure kernel completion and check for errors
87 cudaError_t err = cudaDeviceSynchronize();
88 if (err != cudaSuccess) {
89 std::cerr << "Kernel execution error: " << cudaGetErrorString(err) << std::endl;
90 return 0.0;
91 }
92
93 // Copy result back to host
94 cudaMemcpy(&h_tile_sum, d_partial_sum, sizeof(double),
95 cudaMemcpyDeviceToHost);
96
97 // Accumulate the result
98 total_sum += h_tile_sum;
99 }
100
101 // Apply Simpson's rule scaling factor
102 return total_sum * hx * hy * hz / 27.0;
103 }

118 for (long z_start = 0; z_start < Nz; z_start += tile_size_z) {

124

126 launchSimpson3DKernel(d_f_full + z_start * Nx * Ny, d_partial_sum, Nx, Ny, Nz,
127 tile_size_z, z_start, current_tile_z);
128
129 // Synchronize to ensure kernel completion and check for errors
130 cudaError_t err = cudaDeviceSynchronize();
131 if (err != cudaSuccess) {
132 std::cerr << "Kernel execution error: " << cudaGetErrorString(err) << std::endl;
133 return 0.0;
134 }
135
136 // Copy result back to host
137 cudaMemcpy(&h_tile_sum, d_partial_sum, sizeof(double),
138 cudaMemcpyDeviceToHost);
139
140 // Accumulate the result
141 double temp = total_sum + h_tile_sum;
142 if (std::abs(total_sum) >= std::abs(h_tile_sum)) {
143 compensation += (total_sum - temp) + h_tile_sum;
144 } else {
145 compensation += (h_tile_sum - temp) + total_sum;
146 }
147 total_sum = temp;
148 }
149
150 // Apply Simpson's rule scaling factor
151 return (total_sum + compensation) * hx * hy * hz / 27.0;
152 }
