699void compute_rms_values(const double *d_psi, // Device: 3D psi array
700 double *d_work_array, const double *d_x2,
701 const double *d_y2, const double *d_z2,
702 Simpson3DTiledIntegrator &integ,
703 double *h_rms_pinned) // Output RMS values in pinned memory [rms_x, rms_y, rms_z]clean
704{
705 // Check for CUDA errors
706 cudaError_t err = cudaGetLastError();
707 if (err != cudaSuccess) {
708 printf("CUDA error after memcpy: %s\n", cudaGetErrorString(err));
709 }
710
711 // Configure kernel launch parameters
712 dim3 blockSize(8, 8, 4); // Adjust based on your GPU
713 dim3 gridSize((Nx + blockSize.x - 1) / blockSize.x,
714 (Ny + blockSize.y - 1) / blockSize.y,
715 (Nz + blockSize.z - 1) / blockSize.z);
716
717 // Compute x^2 * psi^2
718 compute_single_weighted_psi_squared<<<gridSize, blockSize>>>(
719 d_psi, d_x2, d_work_array,
720 0 // 0 for x direction
721 );
722
723 cudaDeviceSynchronize();
724 // Integrate x^2 * psi^2
725 double x2_integral =
726 integ.integrateDevice(dx, dy, dz, d_work_array, Nx, Ny, Nz);
727
728 // Compute y^2 * psi^2 (reuse d_work_array)
729 compute_single_weighted_psi_squared<<<gridSize, blockSize>>>(
730 d_psi, d_y2, d_work_array,
731 1 // 1 for y direction
732 );
733 cudaDeviceSynchronize();
734 // Integrate y^2 * psi^2
735 double y2_integral =
736 integ.integrateDevice(dx, dy, dz, d_work_array, Nx, Ny, Nz);
737
738 // Compute z^2 * psi^2 (reuse d_work_array)
739 compute_single_weighted_psi_squared<<<gridSize, blockSize>>>(
740 d_psi, d_z2, d_work_array,
741 2 // 2 for z direction
742 );
743 cudaDeviceSynchronize();
744 // Integrate z^2 * psi^2
745 double z2_integral =
746 integ.integrateDevice(dx, dy, dz, d_work_array, Nx, Ny, Nz);
747
748 // Calculate RMS values and store in pinned memory
749 h_rms_pinned[0] = sqrt(x2_integral); // rms_x
750 h_rms_pinned[1] = sqrt(y2_integral); // rms_y
751 h_rms_pinned[2] = sqrt(z2_integral); // rms_z
752}

761__global__ void compute_single_weighted_psi_squared(
762 const double *__restrict__ psi,
763 const double *__restrict__ coord_squared, // x2, y2, or z2
764 double *result,
765 int direction) // 0=x, 1=y, 2=z
766{
767 int idx = blockIdx.x * blockDim.x + threadIdx.x;
768 int idy = blockIdx.y * blockDim.y + threadIdx.y;
769 int idz = blockIdx.z * blockDim.z + threadIdx.z;
770
771 if (idx >= d_Nx || idy >= d_Ny || idz >= d_Nz)
772 return;
773
774 int linear_idx = idz * d_Ny * d_Nx + idy * d_Nx + idx;
775 double psi_val = __ldg(&psi[linear_idx]); // Read-only cache for psi
776 double psi_squared = psi_val * psi_val;
777
778 double weight = 0.0;
779 if (direction == 0)

781 else if (direction == 1)
782 weight = __ldg(&coord_squared[idy]); // y^2 - read-only cache
783 else if (direction == 2)
784 weight = __ldg(&coord_squared[idz]); // z^2 - read-only cache
785
786 result[linear_idx] = weight * psi_squared;
787}

833void initpsi(double *psi, MultiArray<double> &x2, MultiArray<double> &y2, MultiArray<double> &z2, MultiArray<double> &x, MultiArray<double> &y, MultiArray<double> &z) {
834 long cnti, cntj, cntk;
835 double cpsi;
836 double tmp;
837 cpsi = sqrt(2. * pi * sqrt(2. * pi) * sx * sy * sz);
838
839 #pragma omp parallel for private(cnti)
840 for (cnti = 0; cnti < Nx; cnti++) {
841 x[cnti] = (cnti - Nx2) * dx;
842 x2[cnti] = x[cnti] * x[cnti];
843 }
844
845 #pragma omp parallel for private(cntj)
846 for (cntj = 0; cntj < Ny; cntj++) {

852 for (cntk = 0; cntk < Nz; cntk++) {
853
854 z[cntk] = (cntk - Nz2) * dz;
855 z2[cntk] = z[cntk] * z[cntk];

860 for (cntj = 0; cntj < Ny; cntj++) {
861 for (cnti = 0; cnti < Nx; cnti++) {

863 z2[cntk] / (sz * sz)));

1441void calcmuen(double *muen,double *d_psi, double *d_psi2, double *d_pot, double *d_psi2dd, double *d_potdd, cufftDoubleComplex * d_psi2_fft, cufftHandle forward_plan, cufftHandle backward_plan, Simpson3DTiledIntegrator &integ, const double g, const double gd, const double h2){
1442
1443 dim3 threadsPerBlock(8, 8, 8);
1444 dim3 numBlocks((Nx + threadsPerBlock.x - 1) / threadsPerBlock.x,
1445 (Ny + threadsPerBlock.y - 1) / threadsPerBlock.y,
1446 (Nz + threadsPerBlock.z - 1) / threadsPerBlock.z);
1447
1448 // Step 1: Contact energy - Calculate 0.5 * g * ψ⁴
1449 calcmuen_fused_contact<<<numBlocks, threadsPerBlock>>>(d_psi, d_psi2, g);
