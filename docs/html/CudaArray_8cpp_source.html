41template<typename T>
42CudaArray3D<T>::CudaArray3D(size_t nx_, size_t ny_, size_t nz_, bool use_pitched_memory)
43 : nx(nx_), ny(ny_), nz(nz_), total_elements(nx_ * ny_ * nz_),
44 use_pitched(use_pitched_memory), d_data_linear(nullptr) {
45
46 pitched_ptr = make_cudaPitchedPtr(nullptr, 0, 0, 0);
47
48 if (total_elements > 0) {
49 if (use_pitched && nz > 1 && ny > 1) {
50 // ===== TRUE 3D ALLOCATION (nz > 1, ny > 1) =====
51 // Use cudaMalloc3D for optimal 3D memory layout with hardware alignment
52 cudaExtent extent = make_cudaExtent(nx * sizeof(T), ny, nz);
53 cudaError_t error = cudaMalloc3D(&pitched_ptr, extent);
54
55 if (error != cudaSuccess) {
56 // 3D pitched allocation failed - fallback to linear memory
57 use_pitched = false;
58 size_t bytes = total_elements * sizeof(T);
59 error = cudaMalloc((void**)&d_data_linear, bytes);
60 checkCudaError(error, "Failed to allocate device memory");
61 pitch = nx * sizeof(T); // No actual pitch in linear memory
62 } else {
63 // 3D pitched allocation succeeded
64 pitch = pitched_ptr.pitch;
65 }
66 } else if (use_pitched && ny > 1) {
67 // ===== 2D ALLOCATION (nz = 1, ny > 1) =====
68 // Use cudaMallocPitch for optimal 2D memory layout
69 size_t width = nx * sizeof(T);
70 size_t height = ny * nz; // For 2D: nz=1, so height = ny
71
72 T* ptr2d;
73 cudaError_t error = cudaMallocPitch((void**)&ptr2d, &pitch, width, height);
74
75 if (error != cudaSuccess) {
76 // 2D pitched allocation failed - fallback to linear memory
77 use_pitched = false;
78 size_t bytes = total_elements * sizeof(T);
79 error = cudaMalloc((void**)&d_data_linear, bytes);
80 checkCudaError(error, "Failed to allocate device memory");
81 pitch = nx * sizeof(T); // No actual pitch in linear memory
82 } else {
83 // 2D pitched allocation succeeded - wrap in 3D structure for consistency
84 pitched_ptr = make_cudaPitchedPtr(ptr2d, pitch, nx * sizeof(T), ny);
85 }
86 } else {
87 // ===== LINEAR ALLOCATION =====
88 // For 1D arrays, when pitched memory is not requested, or as fallback
89 use_pitched = false;
90 size_t bytes = total_elements * sizeof(T);
91 cudaError_t error = cudaMalloc((void**)&d_data_linear, bytes);
92 checkCudaError(error, "Failed to allocate device memory");
93 pitch = nx * sizeof(T); // Conceptual pitch for consistency
94 }
95 }

187template<typename T>
188void CudaArray3D<T>::copyFromHost(const T* h_data) {
189 if (h_data == nullptr || total_elements == 0) return; // Safety checks
190
191 if (use_pitched) {
192 // ===== PITCHED MEMORY COPY =====
193 // Use cudaMemcpy3D to handle padding and memory layout automatically
194 cudaMemcpy3DParms params = {0}; // Initialize all fields to zero
195
196 // Configure source (host) - treat as linear memory
197 params.srcPtr = make_cudaPitchedPtr((void*)h_data,
198 nx * sizeof(T), // pitch = width (no padding)
199 nx * sizeof(T), // width in bytes
200 ny); // height
201 params.srcPos = make_cudaPos(0, 0, 0); // Start from origin
202
203 // Configure destination (device) - use pitched memory layout
204 params.dstPtr = pitched_ptr; // Our pitched memory allocation
205 params.dstPos = make_cudaPos(0, 0, 0); // Start from origin
206
207 // Configure copy extent and direction
208 params.extent = make_cudaExtent(nx * sizeof(T), ny, nz); // Full array size
209 params.kind = cudaMemcpyHostToDevice;
210
211 // Execute the 3D copy operation
212 cudaError_t error = cudaMemcpy3D(&params);
213 checkCudaError(error, "Failed to copy data from host to device (pitched)");
214 } else {
215 // ===== LINEAR MEMORY COPY =====
216 // Simple contiguous memory copy - no padding to worry about
217 size_t bytes = total_elements * sizeof(T);
218 cudaError_t error = cudaMemcpy(d_data_linear, h_data, bytes, cudaMemcpyHostToDevice);
219 checkCudaError(error, "Failed to copy data from host to device");
220 }
221}
