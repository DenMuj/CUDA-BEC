41__global__ void diff_kernel(
42 double hx, double hy, double hz,
43 double* __restrict__ psi,
44 double* __restrict__ f_res,
45 long nx, long ny, long nz, int par) {
46
47 int i = threadIdx.x + blockDim.x * blockIdx.x;
48 int j = threadIdx.y + blockDim.y * blockIdx.y;
49 int k = threadIdx.z + blockDim.z * blockIdx.z;
50
51 if (i >= nx || j >= ny || k >= nz) return;
52
53 int idx = k * nx * ny + j * nx + i; // 3D to 1D index
54
55 // Precompute reciprocals
56 const double inv_12hx = 1.0 / (12.0 * hx);
57 const double inv_2hx = 1.0 / (2.0 * hx);
58 const double inv_12hy = 1.0 / (12.0 * hy);
59 const double inv_2hy = 1.0 / (2.0 * hy);
60 const double inv_12hz = 1.0 / (12.0 * hz);
61 const double inv_2hz = 1.0 / (2.0 * hz);
62
63 double dpsidx = 0.0, dpsidy = 0.0, dpsidz = 0.0;
64
65 // X-derivative (∂Ψ/∂x)
66 if (i > 1 && i < nx - 2) {
67 // 4th order central difference
68 int idx_m2 = k * nx * ny + j * nx + (i - 2);
69 int idx_m1 = k * nx * ny + j * nx + (i - 1);
70 int idx_p1 = k * nx * ny + j * nx + (i + 1);
71 int idx_p2 = k * nx * ny + j * nx + (i + 2);
72 dpsidx = (__ldg(&psi[idx_m2]) - 8.0 * __ldg(&psi[idx_m1]) + 8.0 * __ldg(&psi[idx_p1]) - __ldg(&psi[idx_p2])) * inv_12hx;
73 } else if (i == 1) {
74 // 2nd order forward difference
75 int idx_0 = k * nx * ny + j * nx + 0;
76 int idx_2 = k * nx * ny + j * nx + 2;
77 dpsidx = (__ldg(&psi[idx_2]) - __ldg(&psi[idx_0])) * inv_2hx;
78 } else if (i == nx - 2) {
79 // 2nd order backward difference
80 int idx_m2 = k * nx * ny + j * nx + (nx - 3);
81 int idx_0 = k * nx * ny + j * nx + (nx - 1);
82 dpsidx = (__ldg(&psi[idx_0]) - __ldg(&psi[idx_m2])) * inv_2hx;
83 }
84 // Boundary points (i=0, i=nx-1) have dpsidx = 0.0
85
86 // Y-derivative (∂Ψ/∂y)
87 if (j > 1 && j < ny - 2) {
88 // 4th order central difference
89 int idx_m2 = k * nx * ny + (j - 2) * nx + i;
90 int idx_m1 = k * nx * ny + (j - 1) * nx + i;
91 int idx_p1 = k * nx * ny + (j + 1) * nx + i;
92 int idx_p2 = k * nx * ny + (j + 2) * nx + i;
93 dpsidy = (__ldg(&psi[idx_m2]) - 8.0 * __ldg(&psi[idx_m1]) + 8.0 * __ldg(&psi[idx_p1]) - __ldg(&psi[idx_p2])) * inv_12hy;
94 } else if (j == 1) {
95 int idx_0 = k * nx * ny + 0 * nx + i;
96 int idx_2 = k * nx * ny + 2 * nx + i;
97 dpsidy = (__ldg(&psi[idx_2]) - __ldg(&psi[idx_0])) * inv_2hy;
98 } else if (j == ny - 2) {
99 int idx_m2 = k * nx * ny + (ny - 3) * nx + i;
100 int idx_0 = k * nx * ny + (ny - 1) * nx + i;
101 dpsidy = (__ldg(&psi[idx_0]) - __ldg(&psi[idx_m2])) * inv_2hy;
102 }
103
104 // Z-derivative (∂Ψ/∂z)
105 if (k > 1 && k < nz - 2) {
106 // 4th order central difference
107 int idx_m2 = (k - 2) * nx * ny + j * nx + i;
108 int idx_m1 = (k - 1) * nx * ny + j * nx + i;
109 int idx_p1 = (k + 1) * nx * ny + j * nx + i;
110 int idx_p2 = (k + 2) * nx * ny + j * nx + i;
111 dpsidz = (__ldg(&psi[idx_m2]) - 8.0 * __ldg(&psi[idx_m1]) + 8.0 * __ldg(&psi[idx_p1]) - __ldg(&psi[idx_p2])) * inv_12hz;
112 } else if (k == 1) {
113 int idx_0 = 0 * nx * ny + j * nx + i;
114 int idx_2 = 2 * nx * ny + j * nx + i;
115 dpsidz = (__ldg(&psi[idx_2]) - __ldg(&psi[idx_0])) * inv_2hz;
116 } else if (k == nz - 2) {
117 int idx_m2 = (nz - 3) * nx * ny + j * nx + i;
118 int idx_0 = (nz - 1) * nx * ny + j * nx + i;
119 dpsidz = (__ldg(&psi[idx_0]) - __ldg(&psi[idx_m2])) * inv_2hz;
120 }
121
122 // Compute kinetic energy density: |∇Ψ|²
123 double con = 1/(2.0 * (3-par));
124 f_res[idx] = (dpsidx * dpsidx + dpsidy * dpsidy + dpsidz * dpsidz)*con;
125}

169 int idx = k * nx * ny + j * nx + i;

171 // Precompute reciprocals
172 const double inv_12hx = 1.0 / (12.0 * hx);
173 const double inv_2hx = 1.0 / (2.0 * hx);
174 const double inv_12hy = 1.0 / (12.0 * hy);
175 const double inv_2hy = 1.0 / (2.0 * hy);
176 const double inv_12hz = 1.0 / (12.0 * hz);
177 const double inv_2hz = 1.0 / (2.0 * hz);
178
179 // Complex derivatives
180 cuDoubleComplex dpsidx = make_cuDoubleComplex(0.0, 0.0);
181 cuDoubleComplex dpsidy = make_cuDoubleComplex(0.0, 0.0);
182 cuDoubleComplex dpsidz = make_cuDoubleComplex(0.0, 0.0);
183
184 // X-derivative (∂Ψ/∂x) - operating on complex values
185 if (i > 1 && i < nx - 2) {
186 int idx_m2 = k * nx * ny + j * nx + (i - 2);
187 int idx_m1 = k * nx * ny + j * nx + (i - 1);
188 int idx_p1 = k * nx * ny + j * nx + (i + 1);
189 int idx_p2 = k * nx * ny + j * nx + (i + 2);
190
191 // Complex arithmetic: (f[i-2] - 8*f[i-1] + 8*f[i+1] - f[i+2]) / (12*h)
192 cuDoubleComplex term1 = __ldg(&psi[idx_m2]);
193 cuDoubleComplex term2 = cuCmul(make_cuDoubleComplex(8.0, 0.0), __ldg(&psi[idx_m1]));
194 cuDoubleComplex term3 = cuCmul(make_cuDoubleComplex(8.0, 0.0), __ldg(&psi[idx_p1]));
195 cuDoubleComplex term4 = __ldg(&psi[idx_p2]);
196
197 cuDoubleComplex numerator = cuCsub(cuCadd(cuCsub(term1, term2), term3), term4);
198 dpsidx = cuCmul(numerator, make_cuDoubleComplex(inv_12hx, 0.0));
199
200 } else if (i == 1) {
201 int idx_0 = k * nx * ny + j * nx + 0;
202 int idx_2 = k * nx * ny + j * nx + 2;
203 cuDoubleComplex numerator = cuCsub(__ldg(&psi[idx_2]), __ldg(&psi[idx_0]));
204 dpsidx = cuCmul(numerator, make_cuDoubleComplex(inv_2hx, 0.0));
205
206 } else if (i == nx - 2) {
207 int idx_m2 = k * nx * ny + j * nx + (nx - 3);
208 int idx_0 = k * nx * ny + j * nx + (nx - 1);
209 cuDoubleComplex numerator = cuCsub(__ldg(&psi[idx_0]), __ldg(&psi[idx_m2]));
210 dpsidx = cuCmul(numerator, make_cuDoubleComplex(inv_2hx, 0.0));
211 }
212
213 // Y-derivative (∂Ψ/∂y) - similar structure
214 if (j > 1 && j < ny - 2) {
215 int idx_m2 = k * nx * ny + (j - 2) * nx + i;
216 int idx_m1 = k * nx * ny + (j - 1) * nx + i;
217 int idx_p1 = k * nx * ny + (j + 1) * nx + i;
218 int idx_p2 = k * nx * ny + (j + 2) * nx + i;
219
220 cuDoubleComplex term1 = __ldg(&psi[idx_m2]);

223 cuDoubleComplex term4 = __ldg(&psi[idx_p2]);

226 dpsidy = cuCmul(numerator, make_cuDoubleComplex(inv_12hy, 0.0));
227
228 } else if (j == 1) {
229 int idx_0 = k * nx * ny + 0 * nx + i;
230 int idx_2 = k * nx * ny + 2 * nx + i;
231 cuDoubleComplex numerator = cuCsub(__ldg(&psi[idx_2]), __ldg(&psi[idx_0]));
232 dpsidy = cuCmul(numerator, make_cuDoubleComplex(inv_2hy, 0.0));
233
234 } else if (j == ny - 2) {
235 int idx_m2 = k * nx * ny + (ny - 3) * nx + i;
236 int idx_0 = k * nx * ny + (ny - 1) * nx + i;
237 cuDoubleComplex numerator = cuCsub(__ldg(&psi[idx_0]), __ldg(&psi[idx_m2]));
238 dpsidy = cuCmul(numerator, make_cuDoubleComplex(inv_2hy, 0.0));
239 }
240
241 // Z-derivative (∂Ψ/∂z) - similar structure
242 if (k > 1 && k < nz - 2) {
243 int idx_m2 = (k - 2) * nx * ny + j * nx + i;
244 int idx_m1 = (k - 1) * nx * ny + j * nx + i;
245 int idx_p1 = (k + 1) * nx * ny + j * nx + i;
246 int idx_p2 = (k + 2) * nx * ny + j * nx + i;
247
248 cuDoubleComplex term1 = __ldg(&psi[idx_m2]);
249 cuDoubleComplex term2 = cuCmul(make_cuDoubleComplex(8.0, 0.0), __ldg(&psi[idx_m1]));
250 cuDoubleComplex term3 = cuCmul(make_cuDoubleComplex(8.0, 0.0), __ldg(&psi[idx_p1]));
251 cuDoubleComplex term4 = __ldg(&psi[idx_p2]);
252
253 cuDoubleComplex numerator = cuCsub(cuCadd(cuCsub(term1, term2), term3), term4);
254 dpsidz = cuCmul(numerator, make_cuDoubleComplex(inv_12hz, 0.0));
255
256 } else if (k == 1) {
257 int idx_0 = 0 * nx * ny + j * nx + i;
258 int idx_2 = 2 * nx * ny + j * nx + i;
259 cuDoubleComplex numerator = cuCsub(__ldg(&psi[idx_2]), __ldg(&psi[idx_0]));
260 dpsidz = cuCmul(numerator, make_cuDoubleComplex(inv_2hz, 0.0));
261
262 } else if (k == nz - 2) {
263 int idx_m2 = (nz - 3) * nx * ny + j * nx + i;
264 int idx_0 = (nz - 1) * nx * ny + j * nx + i;
265 cuDoubleComplex numerator = cuCsub(__ldg(&psi[idx_0]), __ldg(&psi[idx_m2]));
266 dpsidz = cuCmul(numerator, make_cuDoubleComplex(inv_2hz, 0.0));
267 }
268
269 // Now compute |∇Ψ|² = |∂Ψ/∂x|² + |∂Ψ/∂y|² + |∂Ψ/∂z|²
270 double grad_mag_squared = cuCabs(dpsidx) * cuCabs(dpsidx) +
271 cuCabs(dpsidy) * cuCabs(dpsidy) +
272 cuCabs(dpsidz) * cuCabs(dpsidz);
273
274 double con = 1.0 / (2.0 * (3 - par));
275 f_res[idx] = grad_mag_squared * con;
276 }

281void gauleg(double x1, double x2, MultiArray<double>& x, MultiArray<double>& w) {
282 // Get raw pointers and size from MultiArray
283 double* x_ptr = x.raw();
284 double* w_ptr = w.raw();
285 long N = x.size();
286
287 long m, j, i;
288 double z1, z, xm, xl, pp, p3, p2, p1;
289
290 m = (N + 1) / 2;
291 xm = 0.5 * (x2 + x1);
292 xl = 0.5 * (x2 - x1);
293
294 for(i = 1; i <= m; i++) {
295 z = cos(4. * atan(1.) * (i - 0.25) / (N + 0.5));
296 do {
297 p1 = 1.;
298 p2 = 0.;
299 for(j = 1; j <= N; j++) {
300 p3 = p2;
301 p2 = p1;
302 p1 = ((2. * j - 1.) * z * p2 - (j - 1.) * p3) / j;
303 }
304 pp = N * (z * p1 - p2) / (z * z - 1.);
305 z1 = z;
306 z = z1 - p1 / pp;
307 } while (fabs(z - z1) > GAULEG_EPS);
308
309 // Use raw pointers for array access
310 x_ptr[i] = xm - xl * z;
311 x_ptr[N + 1 - i] = xm + xl * z;
312 w_ptr[i] = 2. * xl / ((1. - z * z) * pp * pp);
313 w_ptr[N + 1 - i] = w_ptr[i];
314 }
315
316 return;
317}

327__host__ double simpint(double h, double *f, long N) {
328 long cnti;
329 double sum, sumi, sumj, sumk;
330
331 sumi = 0.; sumj = 0.; sumk = 0.;
332
333 for(cnti = 1; cnti < N - 1; cnti += 2) {
334 sumi += f[cnti];
335 sumj += f[cnti - 1];
336 sumk += f[cnti + 1];
337 }
338
339 sum = sumj + 4. * sumi + sumk;
340 if(N % 2 == 0) sum += (5. * f[N - 1] + 8. * f[N - 2] - f[N - 3]) / 4.;
341
342 return sum * h / 3.;
343 }
