24__global__ void simpson3d_tiled_reduce(double *f, double *partial_sums,
25 long Nx, long Ny, long Nz,
26 long tile_size_z, long z_start) {
27 extern __shared__ double shared[];
28 double* sum_data = shared;
29 double* comp_data = shared + blockDim.x * blockDim.y * blockDim.z;
30
31 // Calculate thread ID within block
32 long tid = threadIdx.x + threadIdx.y * blockDim.x + threadIdx.z * blockDim.x * blockDim.y;
33
34 // Calculate global 3D indices
35 long idx = blockIdx.x * blockDim.x + threadIdx.x;
36 long idy = blockIdx.y * blockDim.y + threadIdx.y;
37 long idz_global = z_start + blockIdx.z * blockDim.z + threadIdx.z; // Global z index in full volume
38 long idz_local = blockIdx.z * blockDim.z + threadIdx.z; // Local z index within tile
39
40 double local_sum = 0.0;
41
42 // Check bounds and compute weighted value
43 if (idx < Nx && idy < Ny && idz_global < Nz && idz_local < tile_size_z) {
44 // Use local index for accessing the tile data in memory
45 long linear_idx = idz_local * Nx * Ny + idy * Nx + idx;
46
47 // Compute Simpson weights using GLOBAL indices (important for boundary conditions)
48 double weight_x = (idx == 0 || idx == Nx - 1) ? 1.0 : (idx % 2 == 1) ? 4.0 : 2.0;
49 double weight_y = (idy == 0 || idy == Ny - 1) ? 1.0 : (idy % 2 == 1) ? 4.0 : 2.0;
50 double weight_z = (idz_global == 0 || idz_global == Nz - 1) ? 1.0 : (idz_global % 2 == 1) ? 4.0 : 2.0;
51
52 local_sum = f[linear_idx] * weight_x * weight_y * weight_z;
53 }
54
55 // Store in shared memory for reduction
56 sum_data[tid] = local_sum;
57 comp_data[tid] = 0.0;
58 __syncthreads();
59
60 // Perform tree-based reduction in shared memory
61 long block_size = blockDim.x * blockDim.y * blockDim.z;
62 for (unsigned int s = block_size / 2; s > 0; s >>= 1) {
63 if (tid < s && tid + s < block_size) {
64 double val_a = sum_data[tid] + comp_data[tid];
65 double val_b = sum_data[tid + s] + comp_data[tid + s];
66 double temp = val_a + val_b;
67 double corr;
68 if (fabs(val_a) >= fabs(val_b)) {
69 corr = (val_a - temp) + val_b;
70 } else {
71 corr = (val_b - temp) + val_a;
72 }
73 sum_data[tid] = temp;
74 comp_data[tid] = corr;
75 }
76 __syncthreads();
77 }
78
79 // Thread 0 adds the block's sum to global result
80 if (tid == 0) {
81 double block_sum = sum_data[0] + comp_data[0];
82 atomicAdd(&partial_sums[0], block_sum);
83 }
84}
