793__global__ void compute_single_weighted_psi_squared(
794 const cuDoubleComplex *__restrict__ psi,
795 double *result,
796 int direction, // 0=x, 1=y, 2=z
797 const double discretiz)
798{
799 int idx = blockIdx.x * blockDim.x + threadIdx.x;
800 int idy = blockIdx.y * blockDim.y + threadIdx.y;
801 int idz = blockIdx.z * blockDim.z + threadIdx.z;
802
803 if (idx >= d_Nx || idy >= d_Ny || idz >= d_Nz)
804 return;
805
806 int linear_idx = idz * d_Ny * d_Nx + idy * d_Nx + idx;
807 cuDoubleComplex psi_val = __ldg(&psi[linear_idx]); // Read-only cache for psi
808 double psi_squared = psi_val.x * psi_val.x + psi_val.y * psi_val.y;
809
810 double weight = 0.0;
811 if (direction == 0) {
812 // x = idx * scale + (-0.5 * d_Nx * scale)
813 double offset = (-0.5 * static_cast<double>(d_Nx)) * discretiz;
814 double x = fma(static_cast<double>(idx), discretiz, offset);
815 weight = fma(x,x,0.0);
816 } else if (direction == 1) {
817 // y = idy * scale + (-0.5 * d_Ny * scale)
818 double offset = (-0.5 * static_cast<double>(d_Ny)) * discretiz;
819 double y = fma(static_cast<double>(idy), discretiz, offset);
820 weight = fma(y,y,0.0);
821 } else if (direction == 2) {
822 // z = idz * scale + (-0.5 * d_Nz * scale)
823 double offset = (-0.5 * static_cast<double>(d_Nz)) * discretiz;
824 double z = fma(static_cast<double>(idz), discretiz, offset);
825 weight = fma(z,z,0.0);
826 }
827
828 result[linear_idx] = fma(weight, psi_squared, 0.0);
829}
